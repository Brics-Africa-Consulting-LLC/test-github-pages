/**
 * Generates a random number within a specified range.
 * @param {number} min - The minimum value of the range.
 * @param {number} max - The maximum value of the range.
 * @returns {number} A random number within the specified range.
 */
function getRandomNumber(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

// Example usage:
let randomNum = getRandomNumber(1, 100);
console.log(randomNum);


/**
 * Capitalizes the first letter of a string.
 * @param {string} str - The input string.
 * @returns {string} The string with the first letter capitalized.
 */
function capitalizeFirstLetter(str) {
  return str.charAt(0).toUpperCase() + str.slice(1);
}

// Example usage:
let result = capitalizeFirstLetter('hello');
console.log(result); // Output: Hello


/**
 * Checks if a number is even.
 * @param {number} number - The input number.
 * @returns {boolean} Returns true if the number is even, false otherwise.
 */
function isEven(number) {
  return number % 2 === 0;
}

// Example usage:
let num = 10;
console.log(isEven(num)); // Output: true


/**
 * Reverses a given string.
 * @param {string} str - The input string.
 * @returns {string} The reversed string.
 */
function reverseString(str) {
  return str.split('').reverse().join('');
}

// Example usage:
let reversed = reverseString('hello');
console.log(reversed); // Output: olleh


/**
 * Calculates the factorial of a given number.
 * @param {number} n - The input number.
 * @returns {number} The factorial of the input number.
 */
function factorial(n) {
  if (n === 0 || n === 1) {
    return 1;
  }
  return n * factorial(n - 1);
}

// Example usage:
let fact = factorial(5);
console.log(fact); // Output: 120


/**
 * Shuffles the elements of an array.
 * @param {Array} arr - The input array.
 * @returns {Array} A new array with shuffled elements.
 */
function shuffleArray(arr) {
  return arr.slice().sort(() => Math.random() - 0.5);
}

// Example usage:
let originalArray = [1, 2, 3, 4, 5];
let shuffledArray = shuffleArray(originalArray);
console.log(shuffledArray);
